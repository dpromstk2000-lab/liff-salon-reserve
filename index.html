async function renderTimeSlots() { const date = document.getElementById('select_date').value; const selectedVal = document.getElementById('selected_datetime').value; if (!date) return; // 【確実なデータ取得】その日の00:00:00から23:59:59までをピンポイントで取得 const { data: bookings } = await sbClient .from('bookings') .select('start_datetime') .gte('start_datetime', `${date}T00:00:00`) .lte('start_datetime', `${date}T23:59:59`); // その日の予約時間だけを抽出 (例: "10:30") const bookedTimes = bookings ? bookings.map(b => b.start_datetime.substring(11, 16)) : []; const container = document.getElementById('time-slots'); container.innerHTML = ''; for (let h = 10; h < 19; h++) { [0, 30].forEach(m => { const timeStr = `${String(h).padStart(2, '0')}:${String(m).padStart(2, '0')}`; const fullTime = `${date}T${timeStr}:00`; // 【超重要】ボタンを作る際、現在の日付に対する予約かを確認 const isBooked = bookedTimes.includes(timeStr); const btn = document.createElement('button'); btn.innerText = isBooked ? '×' : timeStr; btn.disabled = isBooked; // 選択状態の復元 if (selectedVal === fullTime) btn.classList.add('selected-slot'); btn.onclick = function() { document.querySelectorAll('#time-slots button').forEach(b => b.classList.remove('selected-slot')); this.classList.add('selected-slot'); document.getElementById('selected_datetime').value = fullTime; }; container.appendChild(btn); }); } }
